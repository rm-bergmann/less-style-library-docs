import CodeBlock from '../../components/codeBlock/codeBlock';
import FormSelect from '../../components/formSelect/formSelect';

<div className="code-block">
  <CodeBlock>
    #form.form-element(options);
  </CodeBlock>
</div>

* Useful to use as a starting point when designing the element styles
* Form elements currently supported are **label**, **input**, **select** and **textarea**
* Options may vary depending on the form element

---

## Label

<div className={`code-block__description`}>
  To style a label element use the following mixin. This mixin accepts
  3 parameters with the following default values:

  <p>&nbsp;</p>

  <CodeBlock>
    {`font-size: 1.2rem;
font-weight: 700;
color: @color[primary];`}
  </CodeBlock>
</div>

<div className="code-block">
  <CodeBlock>
    #form.label(@size, @weight, @color);
  </CodeBlock>

  <div className="form-container">
    <label className="form__label--01" htmlFor="text-input">
      Form Label
    </label>
  </div>
</div>

<div className={`code-block__description`}>
  To customize adjust the values, for example:
</div>

<div className="code-block">
  <CodeBlock>
    #form.label(18px, 400, @color[secondary]);
  </CodeBlock>

  <div className="form-container">
    <label className="form__label--02" htmlFor="text-input">
      Form Label
    </label>
  </div>
</div>

## Input

<div className="code-block">
  <CodeBlock>
    #form.input(@type);
  </CodeBlock>
</div>

<div className={`code-block__description`}>
  This library currently includes 4 types of text input styles:
</div>

* Default
* Rounded 
* Dark
* Dark Rounded

<div className="code-block">

      placeholder="Enter Text"

</div>
