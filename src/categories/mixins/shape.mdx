import CodeBlock from '../../components/codeBlock/codeBlock';
import Shape from '../../components/shape/shape';

<div className={`code-block__description`}>
  Shape mixins are constructed based on the following pattern:
</div>

<div className="code-block">
  <CodeBlock mixinName={`shape`}>
    #shape.shape-type(@direction, @size, @color);
  </CodeBlock>
</div>

* Shape-type mixins currently supported are: 
  * **arrow**
  * **round** 
  * **square**
* Direction supports the following values (arrow shapes only) **up**, **down**, **left** or **right**
* Size (width and height as 1 value) in px, em, rem etc
* Color

## Arrow

<div className="code-block">
  <CodeBlock mixinName={`shape`}>
    #shape.arrow(left, 50px, @color[primary]);
  </CodeBlock>

  <Shape index={`01`} />
</div>


<div className="code-block">
  <CodeBlock mixinName={`shape`}>
    #shape.arrow(up, 50px, @cyan);
  </CodeBlock>

  <Shape index={`02`} />
</div>


<div className="code-block">
  <CodeBlock mixinName={`shape`}>
    #shape.arrow(right, 50px, @red);
  </CodeBlock>

  <Shape index={`03`} />
</div>

<div className="code-block">
  <CodeBlock mixinName={`shape`}>

  </CodeBlock>

  <Shape index={`04`} />
</div>

## Round

<div className={`code-block__description`}>

</div>

<div className="code-block">

  <Shape index={`05`} />
