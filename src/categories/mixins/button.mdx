import CodeBlock from '../../components/codeBlock/codeBlock';
import Button from '../../components/button/button';

Button mixins currently supported are:

* flat 
* flat rounded
* flat round
* custom
* link

<div className={`code-block__description`}>
  By default the background colour is set to the @color[primary] setting
  Target the element you want to style and call the mixin. The padding 
  and font-size values are fixed to 1rem. The hover background color 
  slightly darkend for a subtle hover effect. The button text color is fixed
  to white.
</div>

<div className="code-block">

  <div className="code-block__example">
    <CodeBlock mixinName={`button`}>
      #button.flat();
    </CodeBlock>
  </div>

  <Button buttonLabel={`Button 01`} />

</div>

<div className={`code-block__description`}>
  Change the background colour by passing in the hex or color name 
  variable of your choice.
</div>

<div className="code-block">

  <div className="code-block__example">
    <CodeBlock mixinName={`button`}>
      #button.flat(@sea-green);
    </CodeBlock>
  </div>

  <Button buttonLabel={`Button 02`} />

</div>

<div className={`code-block__description`}>
  For rounded edges you can pass in 'rounded' which will automatically
  add a border radius which is set to the @settings[radius] value.
</div>

<div className="code-block">
  <div className="code-block__example">
    <CodeBlock mixinName={`button`}>
      #button.flat(rounded, @plum);
    </CodeBlock>
  </div>

  <Button buttonLabel={`Button 03`} />
</div>

<div className={`code-block__description`}>
  For round buttons pass in 'round' as the first parameter
</div>


<div className="code-block">
  <div className="code-block__example">
    <CodeBlock mixinName={`button`}>
      #button.flat(round, @purple);
    </CodeBlock>
  </div>

  <Button buttonLabel={`Button 04`} />
</div>

<div className={`code-block__description`}>
  The custom button mixins allow you to fully customize a button style in 
  a single line of code.
  The following 5 paramaters are supported, and include the following default values.
  <p>&nbsp;</p>

  <CodeBlock mixinName={`button`}>
    background-color: @color[primary];
  </CodeBlock>
  <CodeBlock mixinName={`button`}>
    color: white;
  </CodeBlock>
  <CodeBlock mixinName={`button`}>
    font-size: 1rem;
  </CodeBlock>
  <CodeBlock mixinName={`button`}>
    padding: 1rem;
  </CodeBlock>
  <CodeBlock mixinName={`button`}>
    border-radius: 0;
  </CodeBlock>
</div>

<div className={`code-block__description`}>
  Here is an example of a custom button using settings variables
</div>

<div className="code-block">
  <div className="code-block__example">
    <CodeBlock mixinName={`button`}>
      #button.custom(@color[pale], @color[primary], 20px, 1.1rem, @settings[radius]);
    </CodeBlock>
    <CodeBlock mixinName={`button`}>
      border: 2px solid @color[primary];
    </CodeBlock>
    <CodeBlock mixinName={`button`}>
      box-shadow: 1px 2px 1px @color[graymedium];
    </CodeBlock>
  </div>

  <Button buttonLabel={`Button 06`} />
</div>

<div className={`code-block__description`}>
  Here is an example of a custom button using random values. To further customize
  this button I style, an additional border and box shadow was added.
</div>

<div className="code-block">
  <div className="code-block__example">
    <CodeBlock mixinName={`button`}>
      #button.custom(rgb(78, 11, 69), #fff, 14px, 1.1rem, 6px);
    </CodeBlock>
  </div>

  <Button buttonLabel={`Button 07`} />
</div>

<div className={`code-block__description`}>
  To style a link as a button use the following line:
</div>

<div className="code-block">
  <div className="code-block__example">
    <CodeBlock mixinName={`button`}>
      #button.link();
    </CodeBlock>
  </div>

  <div className="button-wrapper">
    <a className="button-link" href="#">
      Link 1
    </a>
  </div>
</div>


<div className={`code-block__description`}>
  To customize the link button style pass in the following parameters:
  (Default values are shown in comments)

  <p>&nbsp;</p>

  <CodeBlock mixinName={`button`}>
    background-color /* primary color */
  </CodeBlock>

  <CodeBlock mixinName={`button`}>
    color  /* white */
  </CodeBlock>

  <CodeBlock mixinName={`button`}>
    padding /* .7rem 1rem */
  </CodeBlock>

  <CodeBlock mixinName={`button`}>

  </CodeBlock>

  <CodeBlock mixinName={`button`}>
    border-radius /* 0 */
  </CodeBlock>

</div>
