import CodeBlock from '../../components/codeBlock/codeBlock';
import Gradient from '../../components/gradient/gradient';

## Linear Gradients

<div className={`code-block__description`}>
  Gradient styles are based on the following naming pattern:
</div>

<div className="code-block">
  <CodeBlock>
    #gradient.gradient-type(@direction, @from-color, @to-color)
  </CodeBlock>
</div>

<div className={`code-block__description`}>
  gradient-types currently supported:
</div>

* linear
* radial

<div className={`code-block__description`}>
  @direction is optional, by default it is set as 'to bottom'.
  linear gradient supports the following directions:
</div>

* top  (to top)
* left (to left)
* right (to right)

<div className={`code-block__description`}>
  @from-color is set to @color[primary]
  <br />
  @to-color is set to @color[secondary]
</div>

<div className="code-block">
  <CodeBlock mixinName={`gradient`}>
    #gradient.linear(); 
  </CodeBlock>

  <Gradient index={`01`} />
</div>


<div className={`code-block__description`}>
  Default Linear Gradient (Top)
</div>

<div className="code-block">
  <CodeBlock mixinName={`gradient`}>
    #gradient.linear(top); 
  </CodeBlock>

  <Gradient index={`02`} />
</div>

<div className={`code-block__description`}>
  Default Blue to Right Gradient
</div>

<div className="code-block">
  <CodeBlock>
    #gradient.linear(right, @forest-green, @mint-green);
  </CodeBlock>

  <Gradient index={`03`} />
</div>

<div className={`code-block__description`}>
  Default Orange to Left Gradient
</div>

<div className="code-block">
  <CodeBlock>
    #gradient.linear(left, @forest-green, @mint-green);
  </CodeBlock>

  <Gradient index={`04`} />
</div>


## Radial Gradients

<div className="code-block">
  <CodeBlock>
    #gradient.radial(@from-color, @to-color);
  </CodeBlock>
</div>

<div className={`code-block__description`}>
  For radial gradients use the following mixin:
</div>

<div className="code-block">
  <CodeBlock>
    #gradient.radial();
  </CodeBlock>

  <Gradient index={`05`} />
</div>

<div className={`code-block__description`}>
  To customize the gradient colors pass them in as parameters
</div>

<div className="code-block">
  <CodeBlock>
    #gradient.radial(@apricot, @vermilion);
  </CodeBlock>

  <Gradient index={`06`} />
</div>
